include_directories(${CMAKE_CURRENT_SOURCE_DIR})
foreach( dim 2 3)
  set(modulename "poiseuille")
  set(BENCH "${modulename}_${dim}" )
  set(targetname feel_bench_${BENCH})
  FILE(WRITE ${CMAKE_CURRENT_BINARY_DIR}/${BENCH}.cpp
    "// this file is auto-generated by Feel++ \n\n
#include<poiseuille.hpp>\n\n
int main( int argc, char** argv )
{
    Feel::Assert::setLog( \"${BENCH}.assert\");
    poiseuille<${dim}>( argc, argv );
}
\n" )

add_executable(${targetname} ${BENCH}.cpp )
target_link_libraries(${targetname} feelpp ${FEELPP_LIBRARIES} )

endforeach() # dim
