include_directories(${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR})
add_custom_target(gen_feel_perf_curvature ALL )
#foreach(N 2 3 4 5)
foreach(S "Simplex" )
  #foreach(S "Hypercube")
  foreach(P "Lagrange")
    foreach(D 2)
#      foreach(N 2 3 4 5)
      foreach(N 1 )
        set( LAG "// this file is automatically generated
#include <curvature.hpp>
namespace Feel {
  template class Curvature<${D}, ${P}<${N}>, ${P}<${N}, Vectorial>,  ${S}>\\;
}
")
        OVERWITE_IF_DIFFERENT(gen_feel_perf_curvature "feel_perf_curvature_${S}_${D}D_${P}_${N}.cpp" "${LAG}" curvature.hpp)
        set(lagrange "${CMAKE_CURRENT_BINARY_DIR}/feel_perf_curvature_${S}_${D}D_${P}_${N}.cpp;${lagrange}")
      endforeach()
    endforeach()
  endforeach()
endforeach()

feelpp_add_application( perf_curvature SRCS bench.cpp ${lagrange}  CFG curvature.cfg LINK_LIBRARIES ${FEELPP_LIBRARIES}  DEPS gen_feel_perf_curvature SCRIPTS curvature.tex)
