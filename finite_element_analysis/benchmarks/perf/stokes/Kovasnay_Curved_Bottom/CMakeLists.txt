include_directories(${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR})
add_custom_target(gen_feel_perf_kovasnay_curve ALL )

foreach(GeoOrder 1 2)
       foreach(POrder 1  )
              set( LAG "// this file is automatically generated
#include <stokes_kovasnay_curve.hpp>
namespace Feel {
  template class Stokes_Kovasnay_Curve<${POrder},${GeoOrder}>\\;
}
")
        OVERWITE_IF_DIFFERENT(gen_feel_perf_kovasnay_curve "feel_perf_kovasnay_curve_P${POrder}_G${GeoOrder}.cpp" "${LAG}" stokes_kovasnay_curve.hpp)
        set(taylor_hood2 "${CMAKE_CURRENT_BINARY_DIR}/feel_perf_kovasnay_curve_P${POrder}_G${GeoOrder}.cpp;${taylor_hood2}")
      endforeach()
endforeach()


feelpp_add_application( perf_kovasnay_curve SRCS bench.cpp  ${taylor_hood2} CFG kovasnay_curve.cfg LINK_LIBRARIES ${FEELPP_LIBRARIES}  DEPS  gen_feel_perf_kovasnay_curve)
